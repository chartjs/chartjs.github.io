    let smooth = false;

const actions = [
  {
    name: 'Randomize',
    handler(chart) {
      // Utils.srand(Utils.rand())
      chart.data.datasets.forEach(dataset => {
        dataset.data = generateData();
      });
      chart.update();
    }
  },
  {
    name: 'Smooth',
    handler(chart) {
      smooth = !smooth;
      chart.options.elements.line.tension = smooth ? 0.4 : 0;
      chart.update();
    }
  }
